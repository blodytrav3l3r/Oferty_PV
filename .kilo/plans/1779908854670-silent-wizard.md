# Plan: Przebudowa modułu Rury na 3 fazy (wizard)

## Cel

Podzielić obecny płaski widok `#section-offer` na 3 osobne fazy (kroki kreatora) z wizualnym wskaźnikiem postępu, wzorowane na wizardzie w studniach. Fazy 4-5 (Karta budowy, Zamówienie) są pominięte — będą dodane później.

---

## Fazy

| Faza | Nazwa | Obecny HTML | Docelowo |
|------|-------|-------------|----------|
| 1 | Dane klienta i oferty | Karta `#client-offer-content` (zwijana) | Osobny `.wizard-step` z walidacją przed wyjściem |
| 2 | Wybór produktów | Karta wyszukiwarki + katalog | Osobny `.wizard-step` z pełnym katalogiem + dodawaniem |
| 3 | Oferta (podsumowanie) | Tabela + sticky footer | Osobny `.wizard-step` z tabelą, podsumowaniem, zapisem/eksportem |

Wizard indicator pokazuje 5 kroków (1-5), ale kroki 4 i 5 są **disabled** (szare, bez click).

### 3. `public/js/rury/wizard.js` — NOWY PLIK

```javascript
/* ===== KREATOR OFERTY (RURY) — FAZY 1-3 ===== */

let currentWizardStep = 1;

function goToPhase(step) {
    if (step < 1 || step > 3) return; // phase 4-5 przyszłościowo

    currentWizardStep = step;

    // Przełącz widoczność kroków
    document.querySelectorAll('.rury-wizard-step').forEach((s) => s.classList.remove('active'));
    const target = document.getElementById('rury-wizard-step-' + step);
    if (target) target.classList.add('active');

    updateWizardIndicator();
}

function phaseNext() {
    const next = currentWizardStep + 1;
    if (next > 3) return;

    // Walidacja przed przejściem dalej
    if (!validatePhase(currentWizardStep)) return;

    goToPhase(next);
}

function phasePrev() {
    const prev = currentWizardStep - 1;
    if (prev < 1) return;
    goToPhase(prev);
}

function validatePhase(step) {
    switch (step) {
        case 1:
            // Sprawdź wymagane pola klienta
            const clientName = document.getElementById('client-name').value.trim();
            if (!clientName) {
                showToast('Wpisz nazwę klienta przed przejściem dalej', 'error');
                return false;
            }
            return true;
        case 2:
            // Sprawdź czy jest co najmniej 1 produkt
            if (!currentOfferItems || currentOfferItems.length === 0) {
                showToast('Dodaj co najmniej jeden produkt przed przejściem dalej', 'error');
                return false;
            }
            return true;
        default:
            return true;
    }
}

function updateWizardIndicator() {
    for (let i = 1; i <= 5; i++) {
        const dot = document.querySelector(`.rury-wizard-step-dot[data-step="${i}"]`);
        if (!dot) continue;
        dot.classList.remove('active', 'completed');
        if (i === currentWizardStep) dot.classList.add('active');
        else if (i < currentWizardStep) dot.classList.add('completed');
    }
    // Linie
    for (let i = 1; i <= 4; i++) {
        const line = document.getElementById('rury-wizard-line-' + i);
        if (line) line.classList.toggle('completed', currentWizardStep > i);
    }
}

function initWizard() {
    // Start od fazy 1
    goToPhase(1);

    // Jeśli edytujemy ofertę, idź do fazy 3
    if (editingOfferId) {
        goToPhase(3);
        renderOfferItems();
    }
}

// Globalne aliasy
window.goToPhase = goToPhase;
window.phaseNext = phaseNext;
window.phasePrev = phasePrev;
window.currentWizardStep = currentWizardStep;
```
